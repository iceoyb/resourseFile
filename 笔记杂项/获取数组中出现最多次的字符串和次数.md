题目：获取数组中出现最多次的字符串和次数

```javascript

// 入参
let arr = ["a", "a", "a", "a", "a", "a", "c", "a", "b", "a", "a", "a", "a", "a", "a",]

// 获取字典
const getDicMap = (arr) => {
    let numDic = new Object()
    for (let i = 0; i < arr.length; i++) {
        let curStr = arr[i]
        if (numDic[curStr]) {
            numDic[curStr]++
        } else {
            numDic[curStr] = 1
        }
    }
    return numDic
}

// 获取最大字符串和最大次数方法
const getMaxStrAndTimes = (obj) => {
    let max = 0
    let maxStr
    for (let key in obj) {
        if (obj[key] > max) {
            maxStr = key
            max = obj[key]
        }
    }
    return { maxStr, max }
}

// 获取答案方法
let getRes = () => {
    return getMaxStrAndTimes(getDicMap(arr))
}

console.log(getRes())
```
